#include <ESP8266WiFi.h>
#include <WiFiClient.h>
#include <ThingSpeak.h>
#include "DHT.h"
#include <SoftwareSerial.h>   

// WiFi Credentials
const char* ssid = "IOT";
const char* password = "IOT@123456789";

// ThingSpeak Details
unsigned long myChannelNumber = 2916667;
const char* myWriteAPIKey = "4UQQUICCW438X812";

// DHT11 Sensor
#define DHTPIN 2
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

// Soil Moisture Sensor
#define SOIL_MOISTURE_PIN A0

// pH Sensor (SoftwareSerial)
#define PH_RX 12  // D6 - Sensor TX
#define PH_TX 13  // D7 - Sensor RX
SoftwareSerial pHSerial(PH_RX, PH_TX);

// GSM Module (SoftwareSerial)
#define GSM_RX 4   // D2
#define GSM_TX 5   // D1
SoftwareSerial gsmSerial(GSM_RX, GSM_TX);

// WiFi Client
WiFiClient client;

// ---------- Helper Function to Read pH ---------- //
float readPH() {
  unsigned long start = millis();
  String line;

  while (millis() - start < 2000) { // wait up to 2 seconds
    if (pHSerial.available()) {
      line = pHSerial.readStringUntil('\n');
      line.trim();
      int idx = line.indexOf("PH:");
      if (idx != -1) {
        int comma = line.indexOf(',', idx);
        String num = line.substring(idx + 3, comma);
        num.trim();
        float v = num.toFloat();
        if (v >= 0.0 && v <= 14.0) {
          return v;  // Valid pH value
        }
      }
    }
  }
  return NAN; // return Not-A-Number if invalid or timeout
}

// ---------- Helper Function to Send SMS ---------- //
void sendSMSAlert() {
  gsmSerial.println("AT"); 
  delay(1000);
  gsmSerial.println("AT+CMGF=1"); 
  delay(1000);
  gsmSerial.println("AT+CMGS=\"+918248534794\""); 
  delay(1000);
  gsmSerial.print("MOISTURE ALERT: Soil moisture very low!");
  gsmSerial.write(26); // Ctrl+Z to send SMS
  delay(5000);
}

void setup() {
  Serial.begin(9600);
  gsmSerial.begin(9600);
  pHSerial.begin(9600);

  WiFi.begin(ssid, password);
  ThingSpeak.begin(client);
  dht.begin();

  Serial.println("System Ready...");
}

void loop() {
  // Check WiFi and reconnect if necessary
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("WiFi not connected. Attempting to reconnect...");
    WiFi.begin(ssid, password);
    delay(5000);
  }

  // 1. Read all sensors
  int soilMoisture = analogRead(SOIL_MOISTURE_PIN);
  float humidity = dht.readHumidity();
  float temperature = dht.readTemperature();
  float phValue = readPH();

  // 2. Print to Serial Monitor
  Serial.print("Soil Moisture: "); Serial.println(soilMoisture);
  Serial.print("Humidity: "); Serial.println(humidity);
  Serial.print("Temperature: "); Serial.print(temperature); Serial.println(" °C");
  Serial.print("pH Value: ");
  if (isnan(phValue)) Serial.println("Failed to read");
  else Serial.println(phValue);

  // 3. Upload data to ThingSpeak
  ThingSpeak.setField(1, soilMoisture);
  if (!isnan(humidity)) ThingSpeak.setField(2, humidity);
  if (!isnan(temperature)) ThingSpeak.setField(3, temperature);
  if (!isnan(phValue)) ThingSpeak.setField(4, phValue);

  int status = ThingSpeak.writeFields(myChannelNumber, myWriteAPIKey);
  if (status == 200) {
    Serial.println("ThingSpeak Update Successful");
  } else {
    Serial.print("ThingSpeak Update Failed. HTTP error code: ");
    Serial.println(status);
  }

  // 4. Moisture Alert via SMS
  if (soilMoisture < 5) {
    Serial.println("Soil moisture very low! Sending SMS alert...");
    sendSMSAlert();
  }

  delay(20000); // Wait 20 seconds (ThingSpeak minimum limit)
}
